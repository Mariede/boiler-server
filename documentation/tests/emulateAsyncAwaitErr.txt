
// async/ await
const myFunc = () => {
	return new Promise((resolve, reject) => {
		 setTimeout (
			function() {
				reject('Meu erro fabricado aqui!!!');
			},
			2000
		);
	});
};
await myFunc();

const myFunc = () => {
	return new Promise((resolve, reject) => {
		 setTimeout (
			function() {
				reject('Meu erro fabricado aqui!!!');
			},
			2000
		);
	});
};
myFunc()
.catch (
	err => {
		reject(err);
	}
